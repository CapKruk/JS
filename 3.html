#include <iostream>
#include <vector>
using namespace std;

int main() { 
  setlocale(LC_ALL, "Russian");
  
  vector<int> A = {2,3,5,6,6,7,10,11,20,25};
  
  for (int i = 0; i < 10; i++) cout << A[i] << " ";
  cout << endl;
  
  int t, m;
  cout << "Введите удаляемый элемент" << endl;
  cin >> t;
  
  for (int i = 0; i < 10; i++) {
    if (A[i] == t) {m = i; break;}    
  }
  cout << "Номер удаляемого элемента" << endl << m << endl;
  
  //for (int i = m; i < 10; i++) A[i] = A[i + 1]; - можно и так удалить
 // A.pop_back();
  auto iter = A.cbegin(); 
  A.erase(iter + m);
  cout << "Новый массив" << endl;
  for (int i = 0; i < A.size(); i++) cout << A[i] << " "; 
  
  return 0;    
}
